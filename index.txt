<script>
const app=document.getElementById("app");

let state={hugs:0,punches:0,music:0};

function render(content){
  document.body.onclick=null; // reset clicks
  app.innerHTML=`<div class="fade">${content}</div>`;
  requestAnimationFrame(()=>document.querySelector(".fade").classList.add("show"));
}

function smooth(next){
  const el=document.querySelector(".fade");
  if(!el) return next();
  el.classList.remove("show");
  setTimeout(next,350);
}

const intro=[
"Hey Richa… heard today’s been a bit rough.",
"Yeah… I get that. Some days just feel heavier than they should."
];

let i=0;
function showIntro(){
  const current=intro[i];
  render(`<div class="text">${current}</div>`);
  setTimeout(()=>{
    i++;
    if(i<intro.length) showIntro();
    else showType();
  },2200+(current.length*25));
}

const msg=`I know today might not be your day.

But I just wanted to remind you
that you’re really important.

You deserve soft moments,
warmth,
and a million little reasons to smile.`;

function showType(){
  render(`<div class="text" id="type"></div>`);
  const el=document.getElementById("type");
  const words=msg.split(" ");
  let idx=0;

  function type(){
    if(idx<words.length){
      el.innerHTML+=words[idx]+" ";
      idx++;
      setTimeout(type,180);
    }else{
      setTimeout(()=>{
        sunflower();
        setTimeout(showAdaptiveEnding,1400);
        setTimeout(showContinue,2200);
      },600);
    }
  }
  type();
}

function showAdaptiveEnding(){
  let text="take your time… no rush";
  if(state.punches>12) text="hey… you don’t have to carry all that at once";
  else if(state.hugs>=4) text="you’re allowed to be held like this for a while";
  else if(state.music>=3) text="maybe just stay here a little longer";

  const line=document.createElement("div");
  line.innerText=text;
  line.style.position="fixed";
  line.style.bottom="20px";
  line.style.left="50%";
  line.style.transform="translateX(-50%)";
  line.style.opacity="0";
  line.style.transition="1.5s";
  document.body.appendChild(line);
  setTimeout(()=>line.style.opacity="0.7",200);
}

function sunflower(){
  for(let i=0;i<35;i++) spawnSun();
  const interval=setInterval(()=>spawnSun(),120);
  setTimeout(()=>clearInterval(interval),3500);
}

function spawnSun(){
  const s=document.createElement("div");
  s.className="sun";
  s.innerText="🌻";
  s.style.left=Math.random()*100+"vw";
  s.style.top="-10vh";
  s.style.animationDuration=(4+Math.random()*3)+"s";
  document.body.appendChild(s);
  setTimeout(()=>s.remove(),7000);
}

function showContinue(){
  const b=document.createElement("button");
  b.className="subtle";
  b.innerText="tap to continue";
  b.onclick=()=>smooth(showMenu);
  app.appendChild(b);
}

function showMenu(){
  render(`
  <div class="menu">
    <button onclick="smooth(music)">🎵 music</button>
    <button onclick="smooth(hugs)">🫂 hugs</button>
    <button onclick="smooth(flowers)">🌼 soft things</button>
    <button onclick="smooth(punch)">👊 punch</button>
  </div>`);
}

/* MEDIA ARRAYS */
const hugsArr=[
  "hugs/hug1.gif","hugs/hug2.gif","hugs/hug3.gif"
];

const flowerImgs=[
  "flowers/flower1.png","flowers/flower2.png","flowers/flower3.png"
];

/* FEATURES */
function music(){
  state.music++;
  render(`<div class="text">custom music coming soon 🎶</div>
  <button class="subtle" onclick="smooth(showMenu)">back</button>`);
}

function hugs(){
  state.hugs++;
  const gif=hugsArr[Math.floor(Math.random()*hugsArr.length)];
  render(`<div class="text">come here 🫂</div>
  <img src="${gif}" class="hug-img">
  <button class="subtle" onclick="smooth(showMenu)">back</button>`);
}

function flowers(){
  render(`<div class="text">tap anywhere 🌼</div>
  <button class="subtle" onclick="smooth(showMenu)">back</button>`);

  document.body.onclick=(e)=>{
    const f=document.createElement("img");
    f.src=flowerImgs[Math.floor(Math.random()*flowerImgs.length)];
    f.style.position="absolute";
    f.style.width="50px";
    f.style.left=e.clientX+"px";
    f.style.top=e.clientY+"px";
    f.style.transform="translate(-50%,-50%)";
    f.style.transition="2s";
    document.body.appendChild(f);
    setTimeout(()=>{
      f.style.transform="translate(-50%,-120px)";
      f.style.opacity="0";
    },50);
    setTimeout(()=>f.remove(),2000);
  };
}

const ghostTexts=[ "yeah… let it out","it’s okay, I got you","breathe… you’re okay","go on, I’m listening","ugh, same honestly","excuse me?? absolutely not","that deserved a punch" ];

function punch(){
  render(`<div class="text">go on… let it out 👊</div>
  <div id="punchBox">💢</div>
  <div id="count">0</div>
  <button class="subtle" onclick="smooth(showMenu)">done</button>`);

  let c=0;
  const box=document.getElementById("punchBox");
  let lastClick=0,combo=0;

  box.addEventListener("pointerdown",()=>{
    const now=Date.now(),delta=now-lastClick;
    lastClick=now;
    combo=delta<300?combo+1:1;

    c++; state.punches++;
    document.getElementById("count").innerText=c;

    const intensity=Math.min(Math.max(1,300/Math.max(delta,1))+combo/2,10);

    box.style.transform=`scale(${1+intensity*0.05})`;
    box.style.background=`hsl(${45-intensity*3},90%,${75-intensity*3}%)`;
    setTimeout(()=>box.style.transform="scale(1)",100);

    const rect=box.getBoundingClientRect();
    const g=document.createElement("div");
    g.className="ghost";
    g.innerText=ghostTexts[Math.floor(Math.random()*ghostTexts.length)];

    g.style.left=rect.left+rect.width/2+"px";
    g.style.top=rect.top+rect.height/2+"px";
    g.style.setProperty('--x',(Math.random()*120-60)+'px');
    g.style.setProperty('--y',(Math.random()*-120)+'px');

    document.body.appendChild(g);
    setTimeout(()=>g.remove(),1400);
  });
}

showIntro();
